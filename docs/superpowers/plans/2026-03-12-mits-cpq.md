// lib/mits-cpq/validation.ts
// Step validation for MITS CPQ wizard

import type { MITSStepData, MITSWizardStep, MITSTier } from './types';
import { TIER_ORDER } from './types';

export interface ValidationResult {
  valid: boolean;
  errors: string[];
}

export function validateTierSelection(
  data: MITSStepData['tier_selection'],
  tiers: MITSTier[]
): ValidationResult {
  const errors: string[] = [];

  if (!data?.user_count || data.user_count < 1) {
    errors.push('User count must be at least 1');
  }

  if (!data?.selected_tier) {
    errors.push('Please select a tier');
  } else {
    const tier = tiers.find((t) => t.tier_code === data.selected_tier);
    if (!tier) {
      errors.push('Invalid tier selected');
    } else if (data.user_count > tier.target_users_max) {
      errors.push(
        `${tier.tier_name} supports up to ${tier.target_users_max} users. Consider ${getNextTier(tier.tier_code)}.`
      );
    }
  }

  return { valid: errors.length === 0, errors };
}

export function validateM365Config(
  data: MITSStepData['m365_config']
): ValidationResult {
  const errors: string[] = [];

  if (data?.additional_licences !== undefined && data.additional_licences < 0) {
    errors.push('Additional licences cannot be negative');
  }

  if (data?.primary_domain && !isValidDomain(data.primary_domain)) {
    errors.push('Please enter a valid domain (e.g., company.co.za)');
  }

  return { valid: errors.length === 0, errors };
}

export function validateAddOns(
  data: MITSStepData['add_ons']
): ValidationResult {
  const errors: string[] = [];

  if (data?.selected_modules) {
    for (const mod of data.selected_modules) {
      if (mod.quantity < 1) {
        errors.push(`${mod.module_name} quantity must be at least 1`);
      }
    }
  }

  return { valid: errors.length === 0, errors };
}

export function validatePricing(
  data: MITSStepData['pricing']
): ValidationResult {
  const errors: string[] = [];

  if (data?.discount_percent !== undefined) {
    if (data.discount_percent < 0 || data.discount_percent > 100) {
      errors.push('Discount must be between 0% and 100%');
    }
  }

  if (!data?.contract_term_months || data.contract_term_months < 1) {
    errors.push('Please select a contract term');
  }

  if (data?.requires_approval && !data?.margin_percent) {
    errors.push('Discount requires approval');
  }

  return { valid: errors.length === 0, errors };
}

export function validateCustomer(
  data: MITSStepData['customer']
): ValidationResult {
  const errors: string[] = [];

  if (!data?.company_name?.trim()) {
    errors.push('Company name is required');
  }

  if (!data?.primary_contact?.name?.trim()) {
    errors.push('Contact name is required');
  }

  if (!data?.primary_contact?.email || !isValidEmail(data.primary_contact.email)) {
    errors.push('Valid email is required');
  }

  if (!data?.primary_contact?.phone?.trim()) {
    errors.push('Contact phone is required');
  }

  if (!data?.installation_address?.street?.trim()) {
    errors.push('Installation address is required');
  }

  if (!data?.installation_address?.city?.trim()) {
    errors.push('City is required');
  }

  if (!data?.coverage_checked) {
    errors.push('Please check coverage availability');
  }

  return { valid: errors.length === 0, errors };
}

export function validateReview(
  data: MITSStepData['review']
): ValidationResult {
  const errors: string[] = [];

  if (!data?.terms_accepted) {
    errors.push('Please accept the terms and conditions');
  }

  return { valid: errors.length === 0, errors };
}

export function validateStep(
  step: MITSWizardStep,
  data: MITSStepData,
  tiers: MITSTier[] = []
): ValidationResult {
  switch (step) {
    case 'tier_selection':
      return validateTierSelection(data.tier_selection, tiers);
    case 'm365_config':
      return validateM365Config(data.m365_config);
    case 'add_ons':
      return validateAddOns(data.add_ons);
    case 'pricing':
      return validatePricing(data.pricing);
    case 'customer':
      return validateCustomer(data.customer);
    case 'review':
      return validateReview(data.review);
    default:
      return { valid: true, errors: [] };
  }
}

// Helpers
function isValidEmail(email: string): boolean {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function isValidDomain(domain: string): boolean {
  return /^[a-z0-9]([a-z0-9-]{0,61}[a-z0-9])?(\.[a-z]{2,})+$/i.test(domain);
}

function getNextTier(currentTier: string): string {
  const order = TIER_ORDER[currentTier];
  const next = Object.entries(TIER_ORDER).find(([, o]) => o === order + 1);
  return next ? next[0].replace('MITS_', '') : 'Enterprise';
}

// Pricing calculation for MITS CPQ

import type {
  MITSTier,
  MITSM365Pricing,
  MITSSelectedModule,
  MITSStepData,
} from './types';

export interface PricingInput {
  tier: MITSTier;
  m365Pricing: MITSM365Pricing;
  additionalLicences: number;
  selectedModules: MITSSelectedModule[];
  discountPercent: number;
  contractTermMonths: number;
}

export interface PricingOutput {
  base_tier_price: number;
  additional_m365_price: number;
  add_ons_mrc: number;
  add_ons_nrc: number;
  subtotal: number;
  discount_percent: number;
  discount_amount: number;
  total_mrc: number;
  total_nrc: number;
  contract_term_months: number;
  tcv: number; // Total contract value
  margin_percent: number;
  estimated_cost: number;
  requires_approval: boolean;
}

export function calculateMITSPricing(input: PricingInput): PricingOutput {
  const {
    tier,
    m365Pricing,
    additionalLicences,
    selectedModules,
    discountPercent,
    contractTermMonths,
  } = input;

  // Base tier price (MRC)
  const base_tier_price = tier.retail_price;

  // Additional M365 licences (MRC)
  const additional_m365_price = additionalLicences * tier.m365_additional_rate;

  // Add-ons split by billing type
  let add_ons_mrc = 0;
  let add_ons_nrc = 0;

  for (const mod of selectedModules) {
    if (mod.billing_type === 'once_off') {
      add_ons_nrc += mod.total_price;
    } else {
      add_ons_mrc += mod.total_price;
    }
  }

  // Subtotal (MRC only for discount calculation)
  const subtotal = base_tier_price + additional_m365_price + add_ons_mrc;

  // Apply discount to MRC only
  const discount_amount = (subtotal * discountPercent) / 100;
  const total_mrc = subtotal - discount_amount;
  const total_nrc = add_ons_nrc;

  // Total contract value
  const tcv = total_mrc * contractTermMonths + total_nrc;

  // Cost estimation
  const estimated_cost = calculateEstimatedCost(
    tier,
    m365Pricing,
    additionalLicences,
    selectedModules
  );

  // Margin calculation
  const total_revenue_monthly = total_mrc;
  const margin_percent =
    total_revenue_monthly > 0
      ? ((total_revenue_monthly - estimated_cost) / total_revenue_monthly) * 100
      : 0;

  // Approval required if discount exceeds tier margin threshold
  const requires_approval = discountPercent > 10 || margin_percent < 30;

  return {
    base_tier_price,
    additional_m365_price,
    add_ons_mrc,
    add_ons_nrc,
    subtotal,
    discount_percent: discountPercent,
    discount_amount,
    total_mrc,
    total_nrc,
    contract_term_months: contractTermMonths,
    tcv,
    margin_percent: Math.round(margin_percent * 100) / 100,
    estimated_cost,
    requires_approval,
  };
}

function calculateEstimatedCost(
  tier: MITSTier,
  m365Pricing: MITSM365Pricing,
  additionalLicences: number,
  selectedModules: MITSSelectedModule[]
): number {
  // Base tier cost
  let cost = tier.estimated_direct_cost;

  // Additional M365 at CSP cost
  cost += additionalLicences * m365Pricing.csp_cost;

  // Module costs (monthly only)
  for (const mod of selectedModules) {
    if (mod.billing_type !== 'once_off') {
      // Estimate 50% cost for modules (simplified)
      cost += mod.total_price * 0.5;
    }
  }

  return Math.round(cost * 100) / 100;
}

export function recommendTier(userCount: number, tiers: MITSTier[]): string {
  // Find tier where user count falls within range
  const suitable = tiers.find(
    (t) => userCount >= t.target_users_min && userCount <= t.target_users_max
  );

  if (suitable) return suitable.tier_code;

  // If above all tiers, recommend Enterprise
  const maxTier = tiers.reduce((max, t) =>
    t.target_users_max > max.target_users_max ? t : max
  );

  return maxTier.tier_code;
}

export function getContractDiscounts(
  termMonths: number
): { discount: number; label: string } {
  if (termMonths >= 36) return { discount: 15, label: '36-month: 15% off' };
  if (termMonths >= 24) return { discount: 10, label: '24-month: 10% off' };
  if (termMonths >= 12) return { discount: 5, label: '12-month: 5% off' };
  return { discount: 0, label: 'Month-to-month' };
}

import { createClient } from '@/lib/supabase/server';
import { calculateMITSPricing } from '@/lib/mits-cpq/pricing-calculator';
import type { MITSSelectedModule } from '@/lib/mits-cpq/types';

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const {
      tier_code,
      additional_licences = 0,
      selected_modules = [],
      discount_percent = 0,
      contract_term_months = 12,
    } = body;

    if (!tier_code) {
      return NextResponse.json(
        { error: 'tier_code is required' },
        { status: 400 }
      );
    }

    const supabase = await createClient();

    // Fetch tier and M365 pricing
    const [tierResult, m365Result] = await Promise.all([
      supabase
        .from('mits_tier_catalogue')
        .select('*')
        .eq('tier_code', tier_code)
        .single(),
      supabase
        .from('mits_m365_pricing')
        .select('*')
        .eq('is_active', true),
    ]);

    if (tierResult.error || !tierResult.data) {
      return NextResponse.json({ error: 'Invalid tier' }, { status: 400 });
    }

    const tier = tierResult.data;
    const m365Pricing = m365Result.data?.find(

    );

    if (!m365Pricing) {

        { error: 'M365 pricing not found' },
        { status: 500 }
      );
    }

    const pricing = calculateMITSPricing({
      tier,
      m365Pricing,
      additionalLicences: additional_licences,
      selectedModules: selected_modules as MITSSelectedModule[],
      discountPercent: discount_percent,
      contractTermMonths: contract_term_months,
    });

    return NextResponse.json({ pricing });
  } catch (error) {
    console.error('[MITS-CPQ] Pricing calculation failed:', error);
    return NextResponse.json(
      { error: 'Pricing calculation failed' },

    );

}

export function PricingBreakdown({
  pricing,
  tierName,
  showMargin = false,
}: PricingBreakdownProps) {
  if (!pricing) return null;

  return (
    <div className="rounded-lg border border-slate-200 bg-white p-6">
      <h3 className="mb-4 font-semibold text-slate-900">Pricing Breakdown</h3>

      <div className="space-y-3 text-sm">
        <div className="flex justify-between">
          <span className="text-slate-600">
            {tierName || 'Base'} Tier (MRC)
          </span>
          <span className="font-medium">
            R{pricing.base_tier_price.toLocaleString()}
          </span>
        </div>

        {pricing.additional_m365_price > 0 && (
          <div className="flex justify-between">
            <span className="text-slate-600">Additional M365 Licences</span>
            <span className="font-medium">
              R{pricing.additional_m365_price.toLocaleString()}
            </span>
          </div>
        )}

        {pricing.add_ons_mrc > 0 && (
          <div className="flex justify-between">
            <span className="text-slate-600">Add-Ons (MRC)</span>
            <span className="font-medium">
              R{pricing.add_ons_mrc.toLocaleString()}
            </span>
          </div>
        )}

        <div className="flex justify-between border-t pt-2">
          <span className="text-slate-600">Subtotal</span>
          <span className="font-medium">
            R{pricing.subtotal.toLocaleString()}
          </span>
        </div>

        {pricing.discount_percent > 0 && (
          <div className="flex justify-between text-green-600">
            <span>Discount ({pricing.discount_percent}%)</span>
            <span>-R{pricing.discount_amount.toLocaleString()}</span>
          </div>
        )}

        <div className="flex justify-between border-t pt-2 text-lg font-semibold">
          <span>Monthly Total</span>
          <span className="text-orange">
            R{pricing.total_mrc.toLocaleString()}
          </span>
        </div>

        {pricing.total_nrc > 0 && (
          <div className="flex justify-between text-slate-600">
            <span>Once-Off Charges</span>
            <span>R{pricing.total_nrc.toLocaleString()}</span>
          </div>
        )}

        {showMargin && pricing.margin_percent !== undefined && (
          <div className="mt-4 rounded bg-slate-50 p-3">
            <div className="flex justify-between text-sm">
              <span className="text-slate-600">Estimated Cost</span>
              <span>R{pricing.estimated_cost?.toLocaleString()}</span>
            </div>
            <div className="flex justify-between text-sm font-medium">
              <span>Margin</span>
              <span
                className={
                  pricing.margin_percent >= 30
                    ? 'text-green-600'
                    : 'text-amber-600'
                }
              >
                {pricing.margin_percent}%
              </span>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
